--------------------------------------

  -- CREATE TABLE WITH EVERYTHING

--------------------------------------

CREATE TABLE employee_profile (

    employee_id         NUMBER GENERATED BY DEFAULT AS IDENTITY,

    first_name          VARCHAR2(50) NOT NULL,

    middle_initial      CHAR(1) CHECK (middle_initial IS NULL OR REGEXP_LIKE(middle_initial, '^[A-Z]$')),

    last_name           VARCHAR2(50) NOT NULL,

    email               VARCHAR2(100) UNIQUE,

    phone_number        CHAR(10) CONSTRAINT chk_phone_digits CHECK (phone_number BETWEEN '0000000000' AND '9999999999'),

    salary              NUMBER(8, 2) DEFAULT 50000 CHECK (salary >= 30000),

    hire_date           DATE DEFAULT SYSDATE,

    last_login          TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    timezone_login      TIMESTAMP WITH TIME ZONE,

    local_time_login    TIMESTAMP WITH LOCAL TIME ZONE,

    probation_period    INTERVAL DAY(3) TO SECOND(0),

    contract_duration   INTERVAL YEAR(2) TO MONTH,

    department_id       NUMBER,

   

    -- Out-of-line constraints

    CONSTRAINT pk_employee PRIMARY KEY (employee_id),

    CONSTRAINT fk_dept FOREIGN KEY (department_id) REFERENCES departments(department_id),

    CONSTRAINT uc_emp_name UNIQUE (first_name, last_name),

    CONSTRAINT chk_salary_range CHECK (salary BETWEEN 30000 AND 200000)

);
